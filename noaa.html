Function to adjust for different units in CROPDMGEXP and PROPDMGEXP

transform <- function (x, y) {

        if (x == "H" | x == "h")
                {y <- y * 10
        } else if (x == "K" | x == "k") 
                {y <- y * 10^3
        } else if (x == "M" | x == "m") 
                {y <- y * 10^6
        } else if (x == "B" | x == "b") 
                {y <- y * 10^9
        } else if (x == "-" | x == "?")
                 {y <- 0
        }
        y
}

Use transform to adjust CROPDMGEXP and PROPDMGEXP

econ_df$units_adj_cropdmg <- mapply(transform,
